Dichte der Beta-Verteilung

d3 = require('d3')
mathfn = require('https://bundle.run/mathfn@1.1.0')
import {slider} from "@jashkenas/inputs"
import {palette} from "@herbps10/flat-ui-colors"
viewof alpha = slider({ min: 0.1, max: 10, step: 0.1, title: tex`\alpha` })
viewof beta = slider({ min: 0.1, max: 10, step: 0.1, title: tex`\beta` })


mean = (parameters) => parameters.alpha / (parameters.alpha + parameters.beta)
variance = (parameters) => (parameters.alpha * parameters.beta ) / (Math.pow(parameters.alpha + parameters.beta, 2) * (parameters.alpha + parameters.beta + 1))
domain = [...Array(1000).keys()].slice(1).map((x) => x / 1000)
ydomain = null
parameters = ({ alpha, beta })
density = (x, parameters) => (Math.pow(x, parameters.alpha - 1) * Math.pow(1 - x, parameters.beta - 1)) / mathfn.beta(parameters.alpha, parameters.beta)
round = d3.format("0.2r")
chart = {
  const height = 400
  const width = 1000
  const svg = d3.select(DOM.svg(width, height));
  const margin = { top: 20, right: 20, bottom: 40, left: 120 };
  
  const data = domain.map((x) => ({ x: x, y: density(x, parameters) }));

  const x = d3.scaleLinear()
    .domain([domain[0], domain[domain.length - 1]])  
    .range([margin.left, width - margin.right])
  
  const y = d3.scaleLinear()
    .domain([0, ydomain == null ? d3.max(data.map((d) => d.y)) : ydomain])
    .range([height - margin.bottom, margin.top]);
  
  const line = d3.line()
    .x((d) => x(d.x))
    .y((d) => y(d.y));
  
  svg.selectAll('path')
    .data([data])
    .enter()
    .append('path')
    .attr('d', (d) => line(d))
    .attr('stroke', '#104e8b')
    .attr('fill', 'transparent')
    .attr('stroke-width', 4)
  
  const mean_line = d3.line()
    .x((d) => x(d.x))
    .y((d) => d.y);
  
  const m = mean(parameters);
  const sd = Math.sqrt(variance(parameters));
  
  svg.append("g")
      .attr("class", "axis axis--x")
      .attr("transform", `translate(${-0.5},${height - margin.bottom})`)
      .call(d3.axisBottom(x))
      .selectAll("text")
      .style("font-size", "30px"); 
  
  
  svg.selectAll('path.variance')
    .data([[ { x: Math.max(domain[0], m - sd), y: height - margin.top - 20 }, { x: Math.min(domain[domain.length - 1], m + sd), y: height - margin.top - 20 } ]])
    .enter()
    .append('path')
    .attr('class', 'variance')
    .attr('d', (d) => mean_line(d))
    .attr('stroke', palette.midnight_blue)
    .attr('stroke-width', 4)
    .attr('fill', 'transparent')
  
  svg.selectAll('path.mean')
    .data([[{ x: m, y: height - margin.top - 17 }, { x: m, y: height - margin.top  - 23}]])
    .enter()
    .append('path')
    .attr('class', 'mean')
    .attr('d', (d) => mean_line(d))
    .attr('stroke', palette.silver)
    .attr('stroke-width', 4)
    .attr('fill', 'transparent');
  
  return svg.node()
}

Bedingte Dichte

Für \(X, Y\) stetig, lässt sich eine bedingte Dichte für \(X \vert Y = y\) definieren.

\[ f(x\, \vert\, y) = \frac{f_{(X, Y)}(x, y)}{f_Y(y)} \]

Dann gilt

\[ \mathbb{E}[g(X) \vert Y= y] = \int_\mathbb{R} g(x)\, f(x\, \vert\, y)\, \mathrm{d}x \]
