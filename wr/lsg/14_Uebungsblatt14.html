library(tidyverse)
set.seed(452)

N <- 10000
X <- rnorm(N)
Z <- X * exp(abs(X))

glue::glue('Geschätzter Mittelwert ist {mean(Z) |> round(2)} und liegt damit nahe unter der berechneten 0')
Geschätzter Mittelwert ist 0.02 und liegt damit nahe unter der berechneten 0
glue::glue('Geschätzte Varianz is {var(Z) |> round(2)} und liegt damit nahe an der oberen Schranke {round(10 * exp(2), 2)}')
Geschätzte Varianz is 71.72 und liegt damit nahe an der oberen Schranke 73.89

Teilaufgabe a

N <- 10000
ns <- c(10, 100, 1000, 10000)
p <- 0.7

Zs <- list()
for (k in seq_along(ns)) {
  n <- ns[k]
  Z_n <- (rbinom(N, n, p) - n * p) / sqrt(n * p * (1 - p))
  Zs[k] <- list(tibble(n = n, Z_n = Z_n))
}

bind_rows(Zs) |> 
  ggplot(aes(x = Z_n)) +
  geom_histogram(aes(y = ..density..)) +
  geom_function(fun = dnorm, col = 'dodgerblue2', size = 2) +
  facet_wrap(vars(n), scales = 'free') +
  theme_minimal(base_size = 16)

Teilaufgabe b

bind_rows(Zs) |> 
  ggplot(aes(sample = Z_n)) +
  geom_qq(alpha = 0.25, size = 0.75) +

  facet_wrap(vars(n), scales = 'free') +
  theme_minimal(base_size = 16)
